#5. Faça um Programa que leia um número e exiba o dia correspondente da semana. (1-Domingo, 2- Segunda, etc.), se digitar outro valor deve aparecer valorinválido.

dia = int(input("digite de 1 a 7 para saber o dia da semana: "))

if dia < 1 or dia > 7:
  print("fora do intervalo(1 a 7): ", dia,"valor invalido")
elif dia==1:
  print("Domingo")
elif dia==2:
  print("Segunda-feira")
elif dia==3:
  print("Terça-feira")
elif dia==4:
  print("Quarta-feira")
elif dia==5:
  print("Quita-feira")
elif dia==6:
  print("Sexta-feira")
else:
  print("Sabado")
